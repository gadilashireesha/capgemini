integer literal

public class Main
{
	public static void main(String[] args) {
	    int i=5;
		System.out.println(i);
	}
}


FLOAT LITERAL
public class Main
{
	public static void main(String[] args) {
	    float i=5.6f;
		System.out.println(i);
	}
}


STRING LITERAL
public class Main
{
	public static void main(String[] args) {

	}
}

CHAR LITERAL
public class Main
{
	public static void main(String[] args) {
	    char i='A';

	}
}

BOOLEAN LITERAL
public class Main {
  public static void main(String[] args) {
    int x = 10;
    int y = 9;
    System.out.println(x > y); 
  }
}
